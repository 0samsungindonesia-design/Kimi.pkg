        // Create floating particles

        function createParticles() {

            const container = document.getElementById('particles');

            for (let i = 0; i < 50; i++) {

                const particle = document.createElement('div');

                particle.className = 'particle';

                particle.style.left = Math.random() * 100 + '%';

                particle.style.animationDelay = Math.random() * 20 + 's';

                particle.style.animationDuration = (15 + Math.random() * 10) + 's';

                container.appendChild(particle);

            }

        }

        // Clock functionality

        function updateClock() {

            const now = new Date();

            const timeStr = now.toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit' });

            const dateStr = now.toLocaleDateString('id-ID', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });

           

            document.getElementById('clock').textContent = timeStr;

            document.getElementById('date').textContent = dateStr;

        }

        // Install package simulation

        function installPackage(id) {

            if (installingPackages.has(id)) return;

           

            installingPackages.add(id);

            const pkg = findPackage(id);

           

            // Update UI

            const btn = document.getElementById(`btn-${id}`);

            const progressContainer = document.getElementById(`progress-${id}`);

            const fill = document.getElementById(`fill-${id}`);

            const percent = document.getElementById(`percent-${id}`);

           

            btn.innerHTML = '<span class="loading"></span>';

            progressContainer.classList.add('active');

           

            let progress = 0;

            const interval = setInterval(() => {

                progress += Math.random() * 15;

                if (progress >= 100) {

                    progress = 100;

                    clearInterval(interval);

                   

                    setTimeout(() => {

                        pkg.installed = true;

                        installingPackages.delete(id);

                        showToast(`✓ ${pkg.title} installed successfully!`, 'success');

                        renderPackages();

                        updateInstalledCount();

                    }, 500);

                }

               

                fill.style.width = progress + '%';

                percent.textContent = Math.floor(progress) + '%';

            }, 200);

        }

        // Uninstall package

        function uninstallPackage(id) {

            const pkg = findPackage(id);

            pkg.installed = false;

            showToast(`✓ ${pkg.title} uninstalled`, 'success');

            renderPackages();

            updateInstalledCount();

        }

        // Show package details

        function showDetails(id) {

            const pkg = findPackage(id);

            showToast(`ℹ ${pkg.title}: ${pkg.description.substring(0, 50)}...`, 'success');

        }

        // Find package in database

        function findPackage(id) {

            for (let cat in pkgDatabase) {

                const found = pkgDatabase[cat].find(p => p.id === id);

                if (found) return found;

            }

            return null;

        }

        // Update installed count

        function updateInstalledCount() {

            let count = 0;

            for (let cat in pkgDatabase) {

                count += pkgDatabase[cat].filter(p => p.installed).length;

            }

            document.getElementById('installed-count').textContent = count;

        }

        // Toast notification

        function showToast(message, type = '') {

            const toast = document.getElementById('toast');

            const toastMessage = document.getElementById('toast-message');

           

            toast.className = 'toast ' + type;

            toastMessage.textContent = message;

           

            setTimeout(() => toast.classList.add('show'), 10);

            setTimeout(() => toast.classList.remove('show'), 3000);

        }

        // Keyboard navigation

        document.addEventListener('keydown', (e) => {

            if (e.key === 'ArrowUp' || e.key === 'ArrowDown') {

                const categories = document.querySelectorAll('.category');

                const current = document.querySelector('.category.active');

                let index = Array.from(categories).indexOf(current);

               

                if (e.key === 'ArrowUp') index = (index - 1 + categories.length) % categories.length;

                if (e.key === 'ArrowDown') index = (index + 1) % categories.length;

               

                categories[index].click();

            }

        });
